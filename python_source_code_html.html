# checking and handing missing values with three methods: 
# handle = 0: replace missing values with zero
# handle = 1: replace missing values with forward values
# handle = 2: drop missing values
def check_missvalues(df, handle = 0):
    c = 0
    if df.isnull().sum().sum()==0:
        print('no missing values')
    else:
        c=1
        print(df.isnull().sum())
        if handle == 0:
            df.fillna(0)
        elif handle == 1:
            df.fillna(method == 'pad')
        else:
            df.dropna()
    return c

# correlation checking: features vs features and features vs target cnt
# return important features

def corr_check(df, thresh_1, thresh_2, path):
    if thresh_1 > 0 and thresh_2 > 0 and thresh_1 < 1 and thresh_2 < 1:
        
        # 2D-histogram of the correlation matrix of data
        plt.figure(figsize=(12, 10))
        sns.heatmap(df.corr(), annot = True).get_figure().savefig(path + 'corr_map.png')
        
        # print important features: the features that have a larger than thresh_1 correlation to target cnt
        corr_cnt = df.corr()['cnt'][:-1]
        abs(corr_cnt).sort_values(ascending = False) 
        important_features = corr_cnt[abs(corr_cnt) > thresh_1].sort_values(ascending = False)
        print(pd.DataFrame(important_features))
        
        # output and save the 2D-histogram of correlation matrix with only high correlated (> thresh_2) features pairs marked
        corr_feature = df.drop('cnt', axis=1).corr()
        plt.figure(figsize=(12, 10))
        sns.heatmap(corr_feature[abs( corr_feature)>= thresh_2], vmax=1.0, vmin=-1.0, linewidths=0.1,
                annot=True, annot_kws={"size": 8}, square=True).get_figure().savefig(path +'corr_features_map.png');
    else: 
        print('Error: Threshhold values is not in range 0-1')
    return pd.DataFrame(important_features)
